代码块 + 输出

▼ 策略代码
Python ▶ 运行 
def calculate_returns(prices):
    # 计算收益率
    returns = prices.pct_change()
    return returns.fillna(0)

# 执行回测
returns = calculate_returns(df['close'])
print(f"总收益: {returns.sum():.2%}")
